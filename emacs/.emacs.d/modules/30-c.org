* 10. Toolchain C (Build & Format)

*¿Por qué el estilo "linux"?* El estilo =linux= de Emacs es K&R con indentación de 8 espacios y llaves en la misma línea. Sobreescribimos el offset a 4 para proyectos de aprendizaje donde 8 espacios resultan excesivos en pantallas pequeñas.

*¿Por qué =clangd= como LSP?* =clangd= es el LSP oficial de LLVM, con soporte activo. Entiende =compile_flags.txt= y =compile_commands.json= sin configuración extra.

*¿Por qué Unity para tests?* Unity es C puro, un solo archivo (~600 líneas), sin dependencias. Para enseñar testing en C, importa que el alumno pueda leer el framework completo si quiere.

*¿Por qué Makefile en vez de CMake para proyectos simples?* CMake genera archivos de build que generan el build real — dos capas de indirección que confunden más de lo que ayudan cuando el objetivo es entender la compilación. Con Makefile, =make build= corre exactamente =gcc src/*.c -o build/nombre=, visible sin abstracción.

  :mode (("CMakeLists\\.txt\\'" . cmake-mode)
         ("\\.cmake\\'" . cmake-mode)))

;; --- 3. Sistema de Compilación Inteligente ---
(defun my/find-and-compile ()
  "Busca build.sh, Makefile, CMakeLists.txt o compila el archivo actual.
   Soporte para nob.h (Tsoding style)."
  (interactive)
  (let* ((project-root (or (and (fboundp 'projectile-project-root)
                                (projectile-project-root))
                           default-directory))
         (nob-script   (expand-file-name "nob.c"          project-root))
         (build-script (expand-file-name "build.sh"       project-root))
         (makefile     (expand-file-name "Makefile"       project-root))
         (cmake-file   (expand-file-name "CMakeLists.txt" project-root)))
    (cond
     ;; 1. nob.h (Tsoding)
     ((file-exists-p nob-script)
      (let ((default-directory project-root))
        (compile "cc nob.c -o nob && ./nob")))
     ;; 2. build.sh
     ((file-exists-p build-script)
      (let ((default-directory project-root))
        (compile build-script)))
     ;; 3. Makefile — usa "make run" si el target existe, si no solo compila
     ((file-exists-p makefile)
      (let ((default-directory project-root)
            (has-run (with-temp-buffer
                       (insert-file-contents makefile)
                       (re-search-forward "^run[ \t]*:" nil t))))
        (compile (if has-run
                     (format "make -C \"%s\" run" project-root)
                   (format "make -C \"%s\"" project-root)))))
     ;; 4. CMake genérico
     ((file-exists-p cmake-file)
      (let ((default-directory project-root))
        (compile (format "cmake --build \"%s/build\"" project-root))))
     ;; 5. Archivo suelto
     ((buffer-file-name)
      (save-buffer)
      (let* ((file (file-name-nondirectory buffer-file-name))
             (exe  (file-name-sans-extension file)))
        (compile (format "gcc -Wall -g %s -o %s && ./%s" file exe exe))))
     ;; 6. Fallback
     (t (message "No se encontró build system válido")))))

;; F5 para compilar en C
(add-hook 'c-mode-hook
          (lambda () (local-set-key (kbd "<f5>") #'my/find-and-compile)))

;; Auto-scroll en compilación
(setq compilation-scroll-output t)

(defun my/new-nob-project (name)
  "Crea estructura básica para un proyecto C manejado por nob.h."
  (interactive "sNombre del proyecto: ")
  (let ((project-dir (expand-file-name name (concat (my/mu-projects-base) "c/"))))
    (make-directory project-dir t)
    (make-directory (concat project-dir "/src") t)
    (make-directory (concat project-dir "/build") t)
    (shell-command (format "curl -sLo %s/nob.h https://raw.githubusercontent.com/tsoding/nob.h/main/nob.h" project-dir))
    (with-temp-file (concat project-dir "/nob.c")
      (insert "#define NOB_IMPLEMENTATION\n")
      (insert "#include \"nob.h\"\n\n")
      (insert "int main(int argc, char **argv) {\n")
      (insert "    NOB_GO_REBUILD_URSELF(argc, argv);\n\n")
      (insert "    Nob_Cmd cmd = {0};\n")
      (insert "    nob_cmd_append(&cmd, \"gcc\");\n")
      (insert "    nob_cmd_append(&cmd, \"-Wall\", \"-Wextra\", \"-g\");\n")
      (insert "    nob_cmd_append(&cmd, \"-o\", \"build/main\");\n")
      (insert "    nob_cmd_append(&cmd, \"src/main.c\");\n\n")
      (insert "    nob_log(NOB_INFO, \"Compilando...\");\n")
      (insert "    if (!nob_cmd_run_sync(cmd)) return 1;\n")
      (insert "    nob_log(NOB_INFO, \"OK\");\n")
      (insert "    return 0;\n")
      (insert "}\n"))
    (with-temp-file (concat project-dir "/src/main.c")
      (insert "#include <stdio.h>\n\n")
      (insert "int main(void) {\n")
      (insert (format "    printf(\"Hola desde %s con nob.h!\\n\");\n" name))
      (insert "    return 0;\n")
      (insert "}\n"))
    (with-temp-file (concat project-dir "/.gitignore")
      (insert "build/\nnob\n*.o\n"))
    (with-temp-file (concat project-dir "/.projectile") (insert ""))
    (message "Proyecto nob creado: %s" project-dir)
    (find-file (concat project-dir "/src/main.c"))))

** Consulta de Errores con IA

#+begin_src emacs-lisp
(defun my/consultar-error-ia ()
  "Envía el error de compilación y el código a Ollama."
  (interactive)
  (let* ((codigo (buffer-substring-no-properties (point-min) (point-max)))
         (error-msg (if (get-buffer "*compilation*")
                        (with-current-buffer "*compilation*"
                          (buffer-substring-no-properties (point-min) (point-max)))
                      "No hay buffer de compilación disponible."))
         (prompt (format "Hola. Estoy programando en C.
Mi código es:\n---\n%s\n---\n
El error de compilación es:\n---\n%s\n---\n
No me des la solución directa. Explícame el error de forma pedagógica
y dame pistas para que yo mismo pueda arreglarlo."
                         codigo error-msg)))
    (message "Analizando error con Ollama...")
    (gptel-request prompt
      :system "Eres un mentor de programación C experto y paciente. Tu objetivo es enseñar, no solo corregir."
      :callback (lambda (response info)
                  (if (not response)
                      (message "Error: ¿Está Ollama corriendo?")
                    (with-current-buffer (get-buffer-create "*Consulta IA*")
                      (erase-buffer)
                      (require 'markdown-mode nil t)
                      (when (fboundp 'markdown-mode) (markdown-mode))
                      (insert response)
                      (display-buffer (current-buffer))
                      (message "Explicación lista.")))))))

(global-set-key (kbd "C-c C") #'my/consultar-error-ia)
#+end_src

(defun my/new-c-makefile-project (name)
  "Crea proyecto C con Makefile profesional (build/test/coverage/asan/valgrind)."
  (interactive "sNombre del proyecto: ")
  (let ((project-dir (expand-file-name name (concat (my/mu-projects-base) "c/"))))
    (make-directory (concat project-dir "/src")         t)
    (make-directory (concat project-dir "/include")     t)
    (make-directory (concat project-dir "/build")       t)
    (make-directory (concat project-dir "/tests/unity") t)
    (with-temp-file (concat project-dir "/Makefile")
      (insert (format "PROJECT = %s\nCC = gcc\n" name))
      (insert "CFLAGS  = -Wall -Wextra -std=c11 -g -Iinclude\n\n")
      (insert "SRC    = $(wildcard src/*.c)\n")
      (insert "OBJ    = $(patsubst src/%.c, build/%.o, $(SRC))\n")
      (insert "TARGET = build/$(PROJECT)\n\n")
      (insert "UNITY  = tests/unity/unity.c\n")
      (insert "TESTS  = $(wildcard tests/test_*.c)\n\n")
      (insert ".PHONY: all build test coverage asan ubsan valgrind clean run\n\n")
      (insert "all: build\nbuild: $(TARGET)\n\n")
      (insert "$(TARGET): $(OBJ) | build/\n\t$(CC) $(CFLAGS) -o $@ $^\n\n")
      (insert "build/%.o: src/%.c | build/\n\t$(CC) $(CFLAGS) -c -o $@ $<\n\n")
      (insert "build/:\n\t@mkdir -p $@\n\n")
      (insert "test: $(UNITY) build\n")
      (insert "\t$(CC) $(CFLAGS) $(TESTS) $(UNITY) $(filter-out build/main.o,$(OBJ)) -o build/test_runner\n")
      (insert "\t./build/test_runner\n\n")
      (insert "$(UNITY):\n\t@mkdir -p tests/unity\n")
      (insert "\t@curl -sLo tests/unity/unity.h     https://raw.githubusercontent.com/ThrowTheSwitch/Unity/master/src/unity.h\n")
      (insert "\t@curl -sLo tests/unity/unity.c     https://raw.githubusercontent.com/ThrowTheSwitch/Unity/master/src/unity.c\n")
      (insert "\t@curl -sLo tests/unity/unity_internals.h https://raw.githubusercontent.com/ThrowTheSwitch/Unity/master/src/unity_internals.h\n\n")
      (insert "coverage: CFLAGS += --coverage\ncoverage: clean build test\n")
      (insert "\t@mkdir -p coverage\n")
      (insert "\tlcov -c -d build/ -o coverage/coverage.info --rc lcov_branch_coverage=1\n")
      (insert "\tgenhtml coverage/coverage.info -o coverage/\n\n")
      (insert "asan: CFLAGS += -fsanitize=address,undefined\nasan: clean $(TARGET)\n\t./$(TARGET)\n\n")
      (insert "ubsan: CFLAGS += -fsanitize=undefined\nubsan: clean $(TARGET)\n\t./$(TARGET)\n\n")
      (insert "run: build\n\t./$(TARGET)\n\n")
      (insert "valgrind: build\n\tvalgrind --leak-check=full --show-leak-kinds=all --track-origins=yes ./$(TARGET)\n\n")
      (insert "clean:\n\trm -rf build/ coverage/ *.gcov *.gcda *.gcno\n"))
    (with-temp-file (concat project-dir "/compile_flags.txt")
      (insert "-Iinclude\n-Wall\n-Wextra\n-std=c11\n-g\n"))
    (with-temp-file (concat project-dir "/src/main.c")
      (insert "#include <stdio.h>\n\nint main(void) {\n")
      (insert (format "    printf(\"Hola desde %s\\n\");\n" name))
      (insert "    return 0;\n}\n"))
    (with-temp-file (concat project-dir "/tests/test_main.c")
      (insert "#include \"unity/unity.h\"\n\n")
      (insert "void setUp(void)    {}\nvoid tearDown(void) {}\n\n")
      (insert "void test_ejemplo(void) {\n    TEST_ASSERT_EQUAL(2, 1 + 1);\n}\n\n")
      (insert "int main(void) {\n    UNITY_BEGIN();\n    RUN_TEST(test_ejemplo);\n    return UNITY_END();\n}\n"))
    (with-temp-file (concat project-dir "/.gitignore")
      (insert "build/\ncoverage/\ntests/unity/\n*.o\n*.gcov\n*.gcda\n*.gcno\n"))
    (with-temp-file (concat project-dir "/.projectile") (insert ""))
    (message "Proyecto C creado: %s" project-dir)
    (find-file (concat project-dir "/src/main.c"))))

(defun my/forja-c-status-line ()

  (let ((gcc (executable-find "gcc"))
        (gdb (executable-find "gdb")))
